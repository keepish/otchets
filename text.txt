static (bool IsValid, string Message) ValidatePassword(string password)
{
    if (string.IsNullOrEmpty(password))
        return (false, "Пароль не задан.");
    string pattern = @"^(?=.{8,30}$)(?=.*\d)[A-Za-z\d\p{P}\p{S}]+$";

    if (!Regex.IsMatch(password, pattern))
        return (false, "Пароль не соответствует требованиям.");

    return (true, "OK");
}
