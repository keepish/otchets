double GetDegree(double a, int n)
{
    if (n == 1)
        return n;
    double result = a;
    for (int i = 1; i < n; i++)
        result = result * a;
    if (n < 0)
        result = 1 / result;
    return result;
}
 возвращает результат вычисления an (a – любое, n – целое, результат проверять на
